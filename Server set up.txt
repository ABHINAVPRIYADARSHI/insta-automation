#Login into Oracle signalling VM to set up Telegram Webhook server on port 8443
	ssh -i signla-vm-ssh.key ubuntu@80.225.226.53
# Open the port in the OS-level firewall (iptables) and in the OCI Security List (the cloud network firewall).
# To check current rules of IP table:
    sudo iptables -L INPUT --line-numbers -v -n
# Inserting the rule at line 7 in IPtable to open port 8443 for Telegram Webhook
	sudo iptables -I INPUT 7 -p tcp --dport 8443 -m state --state NEW -j ACCEPT
# This saves the current table to the persistent configuration file so that it does not get reset on system reboot
	sudo netfilter-persistent save
#Also, from Oracle console,  Networking > Virtual Cloud Networks >> Public Subnet >> Security List >> Add port 8443.

#To verify, if port 8443 is opened for any outward request, you may run a python server and try to ping it from browser
	sudo python3 -m http.server 8443

	sudo chown -R ubuntu:ubuntu /etc/letsencrypt/live/

# 2. Give ownership of the 'archive' folder (where the actual files live)
	sudo chown -R ubuntu:ubuntu /etc/letsencrypt/archive/

# 3. Ensure the folders have 'execute' permissions so the user can enter them
	sudo chmod -R 755 /etc/letsencrypt/live/
	sudo chmod -R 755 /etc/letsencrypt/archive/

# Create system service so that it auto restarts after crash/system reboot 
We will use pm2, a node based tool.

sudo apt install nodejs npm -y
sudo npm install pm2 -g
cd ~/insta-automation
We gave our service a name insta-bot.
pm2 start server.py --name "insta-bot" --interpreter ./venv/bin/python3
pm2 save
#This will output a command starting with sudo env PATH.... Copy and paste that specific command into your terminal and run it.
	pm2 startup
pm2 save
================================================
View Real-time Logs: pm2 logs insta-bot

Check Status/CPU/Memory: pm2 list

Restart after code changes: pm2 restart insta-bot

Stop the bot: pm2 stop insta-bot

The Dashboard: Type pm2 monit for a real-time terminal dashboard showing CPU and memory usage.
=================================================================
To handle logs and prevent memory consumption of logs over a period of time
pm2 install pm2-logrotate
# Rotate the log if it exceeds 5MB
pm2 set pm2-logrotate:max_size 5M

# Keep only the last 10 rotated files (older ones are deleted)
pm2 set pm2-logrotate:retain 10

# Compress the old logs (saves ~90% space)
pm2 set pm2-logrotate:compress true

# Force a rotation every day at midnight, even if the file is small
pm2 set pm2-logrotate:rotateInterval '0 0 * * *'
